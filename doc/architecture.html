06. 버그 방지 패턴

❌ 전역 배열 재할당 금지

enemies = []; // ← 기존 참조 파괴! enemies.length = 0; // ✓ 참조 유지

validate.js [2] 검사. 루프 중 배열 교체 시 이미 이터레이션 중인 코드가 빈 배열을 참조해 크래시.

❌ 인덱스 루프 중 사망 유발 금지

enemies[i].takeDamage(...) // ← 루프 중 stage-clear 유발 // 루프 밖에서 처리하거나 플래그 사용

validate.js [1] 검사. takeDamage → die → triggerStageClear → enemies.length=0 → 루프 크래시.

❌ for..of 루프 중 splice 금지

for (const e of enemies) { enemies.splice(...) } for (const e of [...enemies]) { ... } // ✓ 스냅샷

validate.js [3] 검사. 이터레이터 무효화로 인한 스킵/크래시 방지.

⚠️ activeBoss null 체크 필수

activeBoss.hp -= dmg; // ← null 크래시 if (activeBoss) activeBoss.hp -= dmg;

validate.js [7] 검사. 보스 처치 후 activeBoss = null이 되므로 접근 전 반드시 체크.

⚠️ 상태 변경 후 루프 재시작

gameState = STATE_PLAYING; // 루프가 멈춘 채 방치 gameState = STATE_PLAYING;
ensureGameLoopRunning(); // ✓

validate.js [6] 검사. LEVEL_UP/SHOP 등에서 복귀 시 루프가 자동으로 재시작되지 않을 수 있음.

⚠️ pendingBossCurse 플래그

// 보스 처치 시 즉시 저주 모달 → 중복 트리거 pendingBossCurse = true; // stage clear 후 처리

보스 처치와 스테이지 클리어가 동시 발생 시 저주 모달 중복 방지. isFinalBoss면 저주 스킵.

⚠️ setTimeout 내 player null 체크

setTimeout(() => { player.hp += 5; }, 3000); setTimeout(() => { if (player) player.hp += 5; }, 3000);

validate.js [10] 검사. 딜레이 중 게임 종료 시 player = null이 될 수 있음.

⚠️ mines/blackHoles 경계 체크

for (let i = mines.length-1; i >= 0; i--) { ... } // 루프 내: if (i >= mines.length) continue;

validate.js [11] 검사. 루프 중 배열 축소 시 인덱스 범위 초과 방지.

07. 좌표계

월드 좌표

맵 크기: 2500 × 2500 px
